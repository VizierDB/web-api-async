  --network="host" \
  -e VIZIERSERVER_NAME="Container API for Project 60fc09ed" \
  -e VIZIERSERVER_LOG_DIR=/app/data/logs/container \
  -e VIZIERENGINE_DATA_DIR=/app/data \
  -e VIZIERSERVER_PACKAGE_PATH=/app/resources/packages \
  -e VIZIERSERVER_PROCESSOR_PATH=/app/resources/processors \
  -e VIZIERSERVER_SERVER_PORT=5005 \
  -e VIZIERSERVER_SERVER_LOCAL_PORT=5005 \
  -e VIZIERCONTAINER_PROJECT_ID=60fc09ed \
  -e VIZIERCONTAINER_CONTROLLER_URL=http://localhost:5000/vizier-db/api/v1 \
  containerserver
